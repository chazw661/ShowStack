# --- Constants ---
KNOWN_SECTIONS = {'[InName]', '[MixName]', '[MtxName]', '[StName]', '[StMonoName]', '[DCAName]', '[MuteDCAName]'}

# Section header -> Django model name to import into (Plan 03 reads this)
SECTION_TARGET_MAP = {
    'InName':      'ConsoleInput',          # uses 'source' field (not 'name')
    'MixName':     'ConsoleAuxOutput',
    'MtxName':     'ConsoleMatrixOutput',
    'StMonoName':  'ConsoleStereoOutput',   # CL/QL master stereo bus L/R/Mono
}

# Recognized but skipped per R-01/R-03
OUT_OF_SCOPE_SECTIONS = {'DCAName', 'MuteDCAName'}

# Rivage [StName] is special-cased: _AL/_AR -> ConsoleStereoOutput; _BL/_BR -> skip
# CL/QL [StName] (returns): no target model; whole section skipped, logged informational

# --- Pure functions ---
def detect_family(lines: list[str]) -> str:
    """Returns 'cl_ql', 'rivage_pm', or 'unknown'.

    Reads the first 4 stripped non-empty lines:
    - line 1 must be exactly '[Information]'
    - line 2 is the model string ('CL5', 'QL5', 'CS-R5', etc.)
    - line 3 (Rivage only) starts with 'DSP-'
    """

def parse_section_file(uploaded_file, filename: str = '') -> dict:
    """Parse one Yamaha Editor section CSV file.

    Returns:
      {
        'family': 'cl_ql' | 'rivage_pm' | 'unknown',
        'section': 'InName' | ... | None,
        'header_row': [...],
        'rows': [{'key': '_01', 'channel_number': 1, 'name': 'ch 1', 'color': 'Blue', 'icon': 'Dynamic'}, ...],
        'errors': [{'code': str, 'line': int, 'detail': str}, ...],
        'source_filename': str,
      }

    Per RESEARCH § "Per-Row Error Catalog": never raises on a single bad row.
    File-level fatal codes (E_NO_INFORMATION, E_UNKNOWN_FAMILY, E_NO_SECTION) populate `errors` and set section=None.
    """

def parse_upload(uploaded_file, filename: str) -> dict:
    """Top-level entry. Returns:
      {
        'family': 'cl_ql' | 'rivage_pm' | 'unknown',
        'sections': [parse_section_file_result, ...],   # one per section file
        'fatal_error': str | None,                       # set on file-level failure
        'is_zip': bool,
      }

    Branches on zipfile.is_zipfile(uploaded_file): zip -> iterate every .csv member;
    single csv -> wrap result in a 1-element list.

    All files inside a zip MUST report the same family; cross-family zip = fatal_error.
    """

def is_default_row(section: str, family: str, row: dict) -> bool:
    """True iff this row matches the per-section factory default for its channel number.

    NAME defaults (per RESEARCH § "Per-Section Default-Row Rules"):
      InName       : `ch{N:>2d}` for N<100 else `ch{N}`            COLOR=Blue   ICON=Dynamic
      MixName 1-16 : `MX{N:>2d}`                                     COLOR=Orange ICON=Blank
      MixName 17-24: `Fx {N-16}` (CL5 only)                          COLOR=Orange ICON=Effector
      MtxName      : `MT{N:>2d}`                                     COLOR=Orange ICON=Blank
      StMonoName   : `_01`=ST L, `_02`=ST R, `_03`=MONO              COLOR=Orange ICON=Blank
      StName Rivage: `_AL`,`_AR`=`ST A`; `_BL`,`_BR`=`ST B`          COLOR=Orange ICON=Blank
    """
```

Channel-key parse rule (used by parse_section_file):
- `_01` (CL/QL) -> `channel_number=1` via `int(key.lstrip('_'))`
- `_001` (Rivage InName) -> `channel_number=1` (leading zeros stripped by `int()`)
- `_AL`,`_AR`,`_BL`,`_BR` (Rivage StName) -> `channel_number=None`, but `key` preserved verbatim
- `DCA 1`,`Mute 1` (Rivage MuteDCAName) -> parsed but row goes in errors as `no_target_model_v2.0` informational

Per-row error codes (used in `errors` list):
- `E_BAD_KEY`           — key doesn't match section's expected shape
- `E_KEY_OUT_OF_RANGE`  — channel number exceeds family max (e.g. _289 on Rivage InName, _073 on CL InName)
- `E_UNKNOWN_COLOR`     — COLOR is not in YAMAHA_COLOR_CHOICES (fall back to 'Blue', log original)
- `E_COLUMN_COUNT`      — row has < 4 cols (after stripping trailing-comma empty)
- `E_NAME_TOO_LONG`     — NAME > 100 chars (truncate, log)
- `E_DUPLICATE_KEY`     — same key twice in one section (first wins, second logged)
- `W_NO_MODEL_TARGET`   — section recognized but out-of-scope per R-01/R-03 (informational)

Family channel-number maxes:
- CL5 InName=72, MixName=24 (16 mixes + 8 Fx), MtxName=8, StName=16, StMonoName=3, DCAName=16
- QL5 InName=64, MixName=16, MtxName=8, StName=16, StMonoName=3, DCAName=16
- Rivage InName=288, MixName=72, MtxName=36, StName has 4 string keys, MuteDCAName=24 DCA + 12 Mute

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Uploaded bytes -> parser | Bytes arrive from `request.FILES['csv_file'].file`; parser must not crash on malformed input. |
| `.zip` archive -> per-member iteration | Zip members carry filenames; member names could include `../` traversal attempts. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-06 | Denial of Service | Parser called with multi-MB malformed CSV | mitigate | Plan 03 enforces 5 MB file-size cap at form level. Parser additionally caps parsed rows per section at the family max + 10 (e.g. 298 for Rivage InName); rows beyond are dropped with `E_KEY_OUT_OF_RANGE`. |
| T-02-07 | Tampering | Zip-slip via crafted member names | mitigate | When iterating `zipfile.ZipFile.namelist()`, parser ignores any member whose name contains `..` or starts with `/`. Members are decoded in-memory via `zf.open(name)` — no extraction to filesystem, so traversal cannot escape `media/`. |
| T-02-08 | Tampering | UnicodeDecodeError on non-UTF-8 file | mitigate | Parser wraps `TextIOWrapper(..., errors='replace')` — never raises on byte-level garbage; sets fatal_error='E_ENCODING' and returns. |
| T-02-09 | Information disclosure | None — parser is pure | accept | No PII, no DB, no logging of sensitive data. |
</threat_model>

"""Yamaha Editor channel-label CSV import — pure-function parser.

Inverse direction of `planner/utils/yamaha_export.py`. Reads single-section
CSV files OR a `.zip` bundle of multiple section files, classifies the
console family from the [Information] block, parses each section's rows,
and surfaces per-row + file-level errors.

NO Django ORM imports here. The view layer (planner/views.py Phase 2 views)
imports `parse_upload` and threads the result into a `ConsoleImport`
draft + a diff-vs-current-state computation.

Per CONTEXT.md amendments R-01..R-04 (LOCKED):
- DCA sections recognized but skipped (no ConsoleDCA model in v2.0).
- CL/QL [StName] (stereo returns) skipped (no return model).
- Rivage [StName] _BL/_BR skipped (no stereo group B slot in STEREO_CHOICES).
- Upload accepts single .csv OR .zip of section CSVs.

Reference: RESEARCH § "Section-by-Section Parsing Spec",
§ "Per-Section Default-Row Rules", § "Per-Row Error Catalog",
§ "Code Examples".
"""
from __future__ import annotations
import csv
import io
import math
import zipfile
from io import TextIOWrapper


KNOWN_SECTIONS = {'[InName]', '[MixName]', '[MtxName]', '[StName]', '[StMonoName]', '[DCAName]', '[MuteDCAName]'}

# Section header -> Django model name (Plan 03 imports this dict)
SECTION_TARGET_MAP = {
    'InName':     'ConsoleInput',          # writes to .source (NOT .name)
    'MixName':    'ConsoleAuxOutput',
    'MtxName':    'ConsoleMatrixOutput',
    'StMonoName': 'ConsoleStereoOutput',
}

OUT_OF_SCOPE_SECTIONS = {'DCAName', 'MuteDCAName'}

# Per-family per-section channel-number maxes (used by E_KEY_OUT_OF_RANGE check)
FAMILY_LIMITS = {
    'cl_ql': {
        'InName':     72,   # CL5 max; QL5 is 64 but we accept up to 72 then enforce against family-aware max if needed
        'MixName':    24,
        'MtxName':     8,
        'StMonoName':  3,
        'StName':     16,   # CL/QL returns — section recognized but skipped
        'DCAName':    16,
    },
    'rivage_pm': {
        'InName':    288,
        'MixName':    72,
        'MtxName':    36,
        # StName is string-keyed for Rivage; not a numeric limit
    },
}

# Default-row rules (NAME pattern per channel number)
def _name_default_input(n: int) -> str:
    """`ch 1`..`ch 9`, `ch10`..`ch99`, `ch100`..`ch288` (width-2 right-justified for n<100)."""
    return f'ch{n:>2d}' if n < 100 else f'ch{n}'

def _name_default_mix(n: int, family: str) -> str:
    """CL5 rows 17-24 default to `Fx 1`..`Fx 8`; all other mixes default to `MX{n:>2d}`."""
    if family == 'cl_ql' and 17 <= n <= 24:
        return f'Fx {n - 16}'
    return f'MX{n:>2d}'

def _name_default_mtx(n: int) -> str:
    return f'MT{n:>2d}'

def _name_default_dca(n: int) -> str:
    return f'DCA{n:>2d}'

STMONO_DEFAULT_NAMES = {1: 'ST L', 2: 'ST R', 3: 'MONO'}

# Rivage stereo-group defaults
RIVAGE_STNAME_DEFAULT_NAMES = {'_AL': 'ST A', '_AR': 'ST A', '_BL': 'ST B', '_BR': 'ST B'}

# -- Family detection ------------------------------------------------------

def detect_family(lines: list[str]) -> str:
    """Classify a Yamaha Editor CSV by its [Information] block. See RESEARCH § Console Family Detection Rule."""
    stripped = [l.strip() for l in lines[:5] if l.strip()]
    if not stripped or stripped[0] != '[Information]':
        return 'unknown'
    model = stripped[1] if len(stripped) > 1 else ''
    line3 = stripped[2] if len(stripped) > 2 else ''
    if model.startswith(('CL', 'QL')):
        return 'cl_ql'
    if model.startswith('CS-R') or line3.startswith('DSP-'):
        return 'rivage_pm'
    return 'unknown'


# -- Default-row detection -------------------------------------------------

def is_default_row(section: str, family: str, row: dict) -> bool:
    """True iff this parsed row matches the per-section factory default. See RESEARCH § Per-Section Default-Row Rules."""
    n = row.get('channel_number')
    name = row.get('name', '')
    color = row.get('color', '')
    icon = row.get('icon', '')
    key = row.get('key', '')

    if section == 'InName':
        if n is None:
            return False
        return name == _name_default_input(n) and color == 'Blue' and icon == 'Dynamic'

    if section == 'MixName':
        if n is None:
            return False
        if family == 'cl_ql' and 17 <= n <= 24:
            return name == f'Fx {n - 16}' and color == 'Orange' and icon == 'Effector'
        return name == _name_default_mix(n, family) and color == 'Orange' and icon == 'Blank'

    if section == 'MtxName':
        if n is None:
            return False
        return name == _name_default_mtx(n) and color == 'Orange' and icon == 'Blank'

    if section == 'StMonoName':
        if n not in STMONO_DEFAULT_NAMES:
            return False
        return name == STMONO_DEFAULT_NAMES[n] and color == 'Orange' and icon == 'Blank'

    if section == 'StName' and family == 'rivage_pm':
        if key not in RIVAGE_STNAME_DEFAULT_NAMES:
            return False
        return name == RIVAGE_STNAME_DEFAULT_NAMES[key] and color == 'Orange' and icon == 'Blank'

    # CL/QL StName (returns), DCAName, MuteDCAName — out of scope; never "default" in a way that matters
    return False


# -- Section parsing -------------------------------------------------------

YAMAHA_COLORS = {'Off', 'Red', 'Orange', 'Yellow', 'Green', 'Sky Blue', 'Blue', 'Purple', 'Pink', 'White'}

def _parse_channel_key(key: str, section: str, family: str) -> tuple[int | None, str | None]:
    """Returns (channel_number or None, error_code or None).

    Rivage [StName] keys (`_AL`, `_AR`, `_BL`, `_BR`) return (None, None) — caller treats string key.
    Rivage [MuteDCAName] keys (`DCA 1`, `Mute 1`) return (int, None).
    All other sections expect `_NN` or `_NNN` numeric keys.
    """
    if section == 'StName' and family == 'rivage_pm':
        if key in {'_AL', '_AR', '_BL', '_BR'}:
            return None, None
        return None, 'E_BAD_KEY'

    if section == 'MuteDCAName':
        parts = key.split()
        if len(parts) == 2 and parts[0] in {'DCA', 'Mute'}:
            try:
                return int(parts[1]), None
            except ValueError:
                return None, 'E_BAD_KEY'
        return None, 'E_BAD_KEY'

    if not key.startswith('_'):
        return None, 'E_BAD_KEY'
    try:
        return int(key.lstrip('_')), None
    except ValueError:
        return None, 'E_BAD_KEY'


def parse_section_file(uploaded_file, filename: str = '') -> dict:
    """Parse one Yamaha Editor section CSV file (UTF-8, CRLF). Returns the parser result dict.

    `uploaded_file` is a file-like with a `.file` attribute (Django's
    `UploadedFile`) OR a plain binary stream. Caller passes either.

    See RESEARCH § "Code Examples > Parsing a single section file" for the
    canonical sketch and RESEARCH § "Per-Row Error Catalog" for error codes.
    """
    # Support both Django UploadedFile and plain binary streams
    if hasattr(uploaded_file, 'file'):
        binary = uploaded_file.file
        binary.seek(0)
    else:
        binary = uploaded_file
        try:
            binary.seek(0)
        except (AttributeError, io.UnsupportedOperation):
            pass

    result = {
        'family': 'unknown',
        'section': None,
        'header_row': [],
        'rows': [],
        'errors': [],
        'source_filename': filename,
    }

    try:
        text = TextIOWrapper(binary, encoding='utf-8', newline='', errors='replace')
        reader = csv.reader(text)
        raw_rows = list(reader)
    except Exception as exc:
        result['errors'].append({'code': 'E_ENCODING', 'line': 0, 'detail': str(exc)})
        return result

    if not raw_rows:
        result['errors'].append({'code': 'E_NO_INFORMATION', 'line': 0, 'detail': 'empty file'})
        return result

    # Family detection from first 5 raw rows (rebuild as line list)
    raw_lines = [','.join(r) for r in raw_rows[:5]]
    result['family'] = detect_family(raw_lines)

    if result['family'] == 'unknown':
        result['errors'].append({'code': 'E_UNKNOWN_FAMILY', 'line': 0, 'detail': 'no [Information] block or unrecognised model'})
        return result

    # Locate section header
    section_index = None
    for i, r in enumerate(raw_rows):
        if r and r[0].startswith('[') and r[0] != '[Information]':
            if r[0] in KNOWN_SECTIONS:
                result['section'] = r[0].strip('[]')
                section_index = i
                break

    if not result['section']:
        result['errors'].append({'code': 'E_NO_SECTION', 'line': 0, 'detail': 'no recognised section header'})
        return result

    # Out-of-scope sections — record informational, return empty rows
    if result['section'] in OUT_OF_SCOPE_SECTIONS:
        result['errors'].append({
            'code': 'W_NO_MODEL_TARGET',
            'line': 0,
            'detail': f'section {result["section"]} not imported in v2.0',
            'section': result['section'],
        })
        return result

    if section_index + 1 < len(raw_rows):
        result['header_row'] = raw_rows[section_index + 1]
    data_rows = raw_rows[section_index + 2:]

    # Family-specific cap to defend against absurd row counts
    fam_caps = FAMILY_LIMITS.get(result['family'], {})
    section_cap = fam_caps.get(result['section'])

    seen_keys = set()
    for i, row in enumerate(data_rows, start=section_index + 3):  # 1-indexed line numbers
        if not row or not row[0].strip():
            continue
        # Slice first 4 cols; trailing-comma -> 5-elt list. Pitfall 3.
        if len(row) < 4:
            result['errors'].append({'code': 'E_COLUMN_COUNT', 'line': i, 'detail': f'expected 4 cols, got {len(row)}'})
            continue
        key, name, color, icon = row[0], row[1], row[2], row[3]

        channel_number, key_err = _parse_channel_key(key, result['section'], result['family'])
        if key_err:
            result['errors'].append({'code': key_err, 'line': i, 'detail': f'bad key {key!r}'})
            continue

        # Rivage [StName] _BL/_BR — out of scope per R-03; skip with informational
        if result['section'] == 'StName' and result['family'] == 'rivage_pm' and key in {'_BL', '_BR'}:
            result['errors'].append({
                'code': 'W_NO_MODEL_TARGET',
                'line': i,
                'detail': f'rivage stereo group B leg {key} not supported in v2.0',
                'section': result['section'],
            })
            continue

        # Range check (only when channel_number applies)
        if channel_number is not None and section_cap and channel_number > section_cap:
            result['errors'].append({
                'code': 'E_KEY_OUT_OF_RANGE',
                'line': i,
                'detail': f'channel {channel_number} > section max {section_cap}',
            })
            continue

        # Duplicate guard
        key_id = (key, channel_number)
        if key_id in seen_keys:
            result['errors'].append({'code': 'E_DUPLICATE_KEY', 'line': i, 'detail': f'duplicate key {key!r}'})
            continue
        seen_keys.add(key_id)

        # Color whitelist
        if color and color not in YAMAHA_COLORS:
            result['errors'].append({
                'code': 'E_UNKNOWN_COLOR',
                'line': i,
                'detail': f'unknown color {color!r}; falling back to Blue',
            })
            color = 'Blue'

        # Name length
        if len(name) > 100:
            result['errors'].append({'code': 'E_NAME_TOO_LONG', 'line': i, 'detail': f'name truncated from {len(name)} to 100 chars'})
            name = name[:100]

        result['rows'].append({
            'key': key,
            'channel_number': channel_number,
            'name': name,
            'color': color or 'Blue',
            'icon': icon,
        })

    return result


def parse_upload(uploaded_file, filename: str) -> dict:
    """Top-level parse entry. Branches on `zipfile.is_zipfile`. Per R-04 (LOCKED).

    Returns:
      {'family': str, 'sections': [section_result, ...], 'fatal_error': str|None, 'is_zip': bool}
    """
    # Materialize bytes so we can both detect-zip and re-read
    if hasattr(uploaded_file, 'read'):
        data = uploaded_file.read()
    elif hasattr(uploaded_file, 'file'):
        uploaded_file.file.seek(0)
        data = uploaded_file.file.read()
    else:
        return {'family': 'unknown', 'sections': [], 'fatal_error': 'E_ENCODING', 'is_zip': False}

    out = {'family': 'unknown', 'sections': [], 'fatal_error': None, 'is_zip': False}

    if zipfile.is_zipfile(io.BytesIO(data)):
        out['is_zip'] = True
        try:
            with zipfile.ZipFile(io.BytesIO(data)) as zf:
                families = set()
                for name in zf.namelist():
                    # Zip-slip mitigation (T-02-07): reject traversal attempts and non-csv members
                    if '..' in name or name.startswith('/') or not name.lower().endswith('.csv'):
                        continue
                    with zf.open(name) as member:
                        section_result = parse_section_file(member, filename=name)
                    out['sections'].append(section_result)
                    if section_result['family'] != 'unknown':
                        families.add(section_result['family'])
                if len(families) > 1:
                    out['fatal_error'] = 'E_MIXED_FAMILIES'
                    return out
                if len(families) == 1:
                    out['family'] = next(iter(families))
                elif not out['sections']:
                    out['fatal_error'] = 'E_NO_SECTION'
        except zipfile.BadZipFile:
            out['fatal_error'] = 'E_ENCODING'
        return out

    # Single-CSV path
    section_result = parse_section_file(io.BytesIO(data), filename=filename)
    out['sections'].append(section_result)
    out['family'] = section_result['family']
    if section_result['section'] is None and section_result['errors']:
        # File-level fatal — surface to caller for upload-time error banner
        for err in section_result['errors']:
            if err['code'] in {'E_ENCODING', 'E_NO_INFORMATION', 'E_UNKNOWN_FAMILY', 'E_NO_SECTION'}:
                out['fatal_error'] = err['code']
                break
    return out
```
