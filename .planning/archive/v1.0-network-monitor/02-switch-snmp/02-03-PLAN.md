// -- Phase 2: Show Mode --------------------------------------------------
let currentShowMode = localStorage.getItem('nhm_show_mode') || '{{ show_mode|default:"show" }}';

function setShowMode(mode) {
    currentShowMode = mode;
    localStorage.setItem('nhm_show_mode', mode);
    updateShowModeUI(mode);
    fetch('/audiopatch/network-monitor/show-mode/', {
        method: 'POST',
        headers: {'Content-Type': 'application/json', 'X-CSRFToken': csrfToken},
        body: JSON.stringify({mode: mode}),
    }).catch(function(err) { console.warn('[NHM] Show mode save error:', err); });
}

function updateShowModeUI(mode) {
    document.querySelectorAll('.nhm-mode-btn').forEach(function(btn) {
        btn.classList.remove('nhm-mode-btn--active-show', 'nhm-mode-btn--active-amber');
        var pressed = btn.dataset.mode === mode;
        btn.setAttribute('aria-pressed', pressed ? 'true' : 'false');
        if (pressed) {
            btn.classList.add(mode === 'show' ? 'nhm-mode-btn--active-show' : 'nhm-mode-btn--active-amber');
        }
    });
    var banner = document.getElementById('nhm-mode-banner');
    var bannerText = document.getElementById('nhm-mode-banner-text');
    if (mode === 'show') {
        banner.classList.remove('nhm-mode-banner--visible');
    } else {
        banner.classList.add('nhm-mode-banner--visible');
        var modeLabel = mode.charAt(0).toUpperCase() + mode.slice(1);
        bannerText.textContent = modeLabel + ' mode \u2014 non-critical alerts suppressed';
    }
}

function saveSettings() {
    var cs = document.getElementById('nhm-community-string').value.trim();
    var errEl = document.getElementById('nhm-settings-save-error');
    errEl.style.setProperty('display', 'none', 'important');

    fetch('/audiopatch/network-monitor/snmp-settings/', {
        method: 'POST',
        headers: {'Content-Type': 'application/json', 'X-CSRFToken': csrfToken},
        body: JSON.stringify({community_string: cs}),

    .then(function(r) { return r.json(); })
    .then(function(data) {
        if (data.ok) {
            closeSettings();
            var gear = document.querySelector('.nhm-settings-btn');
            gear.style.setProperty('border-color', 'var(--accent-green)', 'important');

        } else {
            errEl.textContent = data.error || "Couldn't save settings. Check your connection and try again.";
            errEl.style.setProperty('display', 'block', 'important');
        }
    })
    .catch(function() {
        errEl.textContent = "Couldn't save settings. Check your connection and try again.";

function addSwitch() {
    var ip = document.getElementById('nhm-switch-ip').value.trim();
    var label = document.getElementById('nhm-switch-label').value.trim();
    var errEl = document.getElementById('nhm-switch-ip-error');
    errEl.style.setProperty('display', 'none', 'important');
    if (!ip) {
        errEl.textContent = 'Enter a valid IP address.';
        errEl.style.setProperty('display', 'block', 'important');
        return;
    }
    fetch('/audiopatch/network-monitor/add-switch/', {
        method: 'POST',
        headers: {'Content-Type': 'application/json', 'X-CSRFToken': csrfToken},
        body: JSON.stringify({ip: ip, label: label}),
    })
    .then(function(r) { return r.json(); })
    .then(function(data) {

            errEl.style.setProperty('display', 'block', 'important');
        }
    })
    .catch(function() {
        errEl.textContent = "Couldn't add switch. Check your connection.";

    });
}

function refreshSwitchList() {
    fetch('/audiopatch/network-monitor/status/')
    .then(function(r) { return r.json(); })
    .then(function(data) {
        if (!data.ok) return;
        var list = document.getElementById('nhm-switch-list');
        var empty = document.getElementById('nhm-switch-list-empty');
        var switches = (data.devices || []).filter(function(d) { return d.domain === 'switch'; });
        list.innerHTML = '';
        if (switches.length === 0) {
            empty.style.setProperty('display', 'block', 'important');
        } else {
            empty.style.setProperty('display', 'none', 'important');
            switches.forEach(function(sw) {
                var li = document.createElement('li');
                li.className = 'nhm-settings-list-item';
                li.innerHTML = '<span>' + (sw.label || sw.ip) + ' <span style="color:var(--text-dim);font-size:12px">' + sw.ip + '</span></span>' +
                    '<button class="nhm-settings-list-remove" onclick="removeSwitch(' + sw.device_id + ',\'' + (sw.label || sw.ip) + '\')">Remove</button>';
                list.appendChild(li);
            });
        }
    });
}

function removeSwitch(deviceId, name) {
    if (!confirm('Remove ' + name + '? This switch will no longer be monitored by SNMP.')) return;
    fetch('/audiopatch/network-monitor/devices/' + deviceId + '/remove/', {
        method: 'POST',
        headers: {'Content-Type': 'application/json', 'X-CSRFToken': csrfToken},
        body: JSON.stringify({}),
    })
    .then(function(r) { return r.json(); })
    .then(function(data) {
        if (data.ok) refreshSwitchList();
    });
}

function updateSwitchCards(switchPorts, snmpConfiguredFlag) {
    snmpConfigured = snmpConfiguredFlag;
    document.querySelectorAll('.nhm-card[data-domain="switch"]').forEach(function(card) {

            if (arrow) {
                header.insertBefore(summaryEl, arrow);
            } else {
                var arrowSpan = document.createElement('span');
                arrowSpan.className = 'nhm-card-expand-arrow';
                arrowSpan.textContent = '\u25BC';
                header.appendChild(summaryEl);
                header.appendChild(arrowSpan);
            }
            var latency = header.querySelector('.nhm-card-latency');
            if (latency) latency.style.setProperty('display', 'none', 'important');
        }
        if (!snmpConfigured) {
            summaryEl.textContent = '';
            detailInner.innerHTML = '<div class="nhm-port-table--unconfigured"><h4>SNMP not configured</h4><p>Enter a community string in Monitor Settings to see port data.</p><button class="nhm-btn--ghost" onclick="openSettings()">Open Settings</button></div>';
            return;
        }
        if (!ports || ports.length === 0) {
            var devStatus = card.dataset.status;
            if (devStatus !== 'unreachable' && devStatus !== 'unknown') {
                summaryEl.textContent = 'SNMP unreachable';
                summaryEl.style.setProperty('color', 'var(--accent-amber)', 'important');
            } else {
                summaryEl.textContent = '';
            }
            detailInner.innerHTML = '<div class="nhm-port-table--unconfigured"><h4>SNMP unreachable</h4><p>Check the community string in Monitor Settings.</p><button class="nhm-btn--ghost" onclick="openSettings()">Open Settings</button></div>';
            return;
        }
        var upCount = 0, errCount = 0;
        ports.forEach(function(p) {
            if (p.oper_status === 'up') upCount++;
            if (p.error_count > 0) errCount++;
        });
        summaryEl.textContent = ports.length + ' ports \u2014 ' + upCount + ' up \u2014 ' + errCount + ' err';
        summaryEl.style.removeProperty('color');
        var existingTable = detailInner.querySelector('.nhm-port-table');
        if (!existingTable) {
            var wrap = document.createElement('div');
            wrap.className = 'nhm-port-table-wrap';
            var table = document.createElement('table');
            table.className = 'nhm-port-table';
            table.innerHTML = '<thead><tr><th scope="col">#</th><th scope="col">Status</th><th scope="col">Speed</th><th scope="col">Bandwidth</th></tr></thead><tbody></tbody>';
            wrap.appendChild(table);
            detailInner.innerHTML = '';
            detailInner.appendChild(wrap);
            existingTable = table;
        }
        var tbody = existingTable.querySelector('tbody');
        ports.forEach(function(p, i) {
            var row = tbody.rows[i];
            if (!row) {
                row = tbody.insertRow();
                row.className = 'nhm-port-row';
                row.innerHTML = '<td class="nhm-port-cell nhm-port-cell--num"></td><td class="nhm-port-cell nhm-port-cell--status"></td><td class="nhm-port-cell nhm-port-cell--speed"></td><td class="nhm-port-cell nhm-port-cell--bw"></td>';
            }
            row.title = 'Errors: ' + (p.error_count || 0) + ' since session start';
            row.cells[0].textContent = p.port_index;
            var dotClass = p.oper_status === 'up' ? 'nhm-dot--online' : p.oper_status === 'down' ? 'nhm-dot--offline' : 'nhm-dot--unknown';
            row.cells[1].innerHTML = '<span class="nhm-dot ' + dotClass + '" aria-label="' + p.port_index + ': ' + p.oper_status + '"></span>';
            var speed = p.speed_mbps;
            var speedText = '\u2014';
            if (speed === 10000) speedText = '10G';
            else if (speed === 1000) speedText = '1G';
            else if (speed === 100) speedText = '100M';
            else if (speed === 10) speedText = '10M';
            else if (speed && speed > 0) speedText = speed + 'M';
            row.cells[2].textContent = speedText;
            var bw = p.bandwidth_pct;
            if (bw !== null && bw !== undefined) {
                var bwClass = bw > 90 ? 'nhm-bw--crit' : bw > 70 ? 'nhm-bw--warn' : 'nhm-bw--ok';
                var srLabel = bw > 90 ? 'critical' : bw > 70 ? 'warning' : 'ok';
                row.cells[3].innerHTML = '<span class="' + bwClass + '">' + bw + '%</span><span class="sr-only">(' + srLabel + ')</span>';
            } else {
                row.cells[3].textContent = '\u2014';
            }
        });
        while (tbody.rows.length > ports.length) {
            tbody.deleteRow(tbody.rows.length - 1);
        }
    });
}
